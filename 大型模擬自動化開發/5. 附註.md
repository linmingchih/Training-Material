# 附註


### FastAPI
#### 服務
```python
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel

app = FastAPI(title="簡單四則運算服務")

# 定義請求資料的格式（Schema）
class CalculationRequest(BaseModel):
    num1: float
    num2: float
    operation: str  # 支援: add, sub, mul, div

@app.post("/calculate")
def calculate(request: CalculationRequest):
    op = request.operation.lower()
    
    if op == "add":
        result = request.num1 + request.num2
    elif op == "sub":
        result = request.num1 - request.num2
    elif op == "mul":
        result = request.num1 * request.num2
    elif op == "div":
        if request.num2 == 0:
            raise HTTPException(status_code=400, detail="除數不能為零！")
        result = request.num1 / request.num2
    else:
        raise HTTPException(status_code=400, detail="不支援的運算符號，請使用 add, sub, mul, 或 div")
        
    return {
        "formula": f"{request.num1} * {request.num2}" if op == "mul" else f"{request.num1} {op} {request.num2}",
        "result": result
    }

if __name__ == "__main__":
    import uvicorn
    # 啟動服務，執行在本地端的 8000 連接埠
    uvicorn.run(app, host="127.0.0.1", port=8000)

```

#### 請求
```python
import requests

# FastAPI 服務的網址與路徑
url = "http://127.0.0.1:8000/calculate"

# 準備要傳送的資料（計算 3 * 3）
# operation 可換成 "add"(加), "sub"(減), "mul"(乘), "div"(除)
data = {
    "num1": 10,
    "num2": 3,
    "operation": "mul"
}

print("正在發送 HTTP POST 請求給 FastAPI...")

# 發送請求，並以 JSON 格式傳遞資料
response = requests.post(url, json=data)

# 檢查是否連線成功
if response.status_code == 200:
    # 解析回傳的 JSON 結果
    result_data = response.json()
    print("\n--- 計算成功 ---")
    print(f"計算公式: {result_data['formula']}")
    print(f"計算結果: {result_data['result']}")
else:
    print(f"\n錯誤！伺服器回應狀態碼: {response.status_code}")
    print(f"錯誤訊息: {response.json()}")

```
